// theme.js

export const COLORS = {
  primary: "#0B3A78",      // Navy Blue (from logo)
  secondary: "#000000",    // Black
  background: "#FFFFFF",   // White
  surface: "#F8F9FA",
  text: "#1A1A1A",
  textLight: "#6B7280",
  border: "#E5E7EB",
  success: "#22C55E",
  warning: "#F59E0B",
  error: "#EF4444",
};

export const FONTS = {
  regular: "System",
  medium: "System",
  bold: "System",
};

export const FONT_SIZE = {
  xs: 12,
  sm: 14,
  md: 16,
  lg: 20,
  xl: 24,
  xxl: 32,
};

export const SPACING = {
  xs: 4,
  sm: 8,
  md: 16,
  lg: 24,
  xl: 32,
  xxl: 40,
};

export const BORDER_RADIUS = {
  sm: 6,
  md: 10,
  lg: 16,
  xl: 24,
  round: 999,
};

export const SHADOW = {
  shadowColor: "#000",
  shadowOffset: {
    width: 0,
    height: 2,
  },
  shadowOpacity: 0.1,
  shadowRadius: 4,
  elevation: 3,
};

const theme = {
  COLORS,
  FONTS,
  FONT_SIZE,
  SPACING,
  BORDER_RADIUS,
  SHADOW,

export default theme;
